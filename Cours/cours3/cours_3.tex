 \frametitle{Notion de tableau}
\begin{itemize}
\item Pour pouvoir écrire des programmes un peu plus intéressants, il est nécessaire d'appliquer des traitements à des séquences de données. Une manière classique est de les rassembler dans un \textbf{\textcolor{NavyBlue}{tableau}}.
\vspace{0.5cm}
\item Dans l'exemple qui suit, on réunit dans un tableau le nombre d'heures d'ensoleillement de chaque jour de la semaine. 
\end{itemize}
\vspace{0.5cm}
\begin{center}
\begin{tabular}{|c|c|c|c|c|c|c|}
 \hline 
\texttt{\hspace{0.2cm}4\hspace{0.2cm}} & \texttt{\hspace{0.2cm}8\hspace{0.2cm}} &\texttt{\hspace{0.2cm}2\hspace{0.2cm}} &\texttt{\hspace{0.2cm}6\hspace{0.2cm}} &\texttt{\hspace{0.2cm}9\hspace{0.2cm}} &\texttt{\hspace{0.2cm}3\hspace{0.2cm}} &\texttt{\hspace{0.2cm}1\hspace{0.2cm}}

\begin{frame}[fragile]
 \frametitle{Les tableaux }
\begin{itemize}
\item Un \textbf{\textcolor{NavyBlue}{tableau}} est une \textbf{\textcolor{NavyBlue}{structure de données}} qui réunit des valeurs (données) d'un \textbf{\textcolor{NavyBlue}{même type}} (le type \texttt{\textcolor{NavyBlue}{int}} dans l'exemple). 

\vspace{0.2cm}
\item C'est une suite de cases contiguës repérées (indicées) par des entiers (\texttt{\textcolor{NavyBlue}{int}}), en partant \texttt{\textcolor{NavyBlue}{0}}.

\vspace{0.2cm}
\item Un tableau constitue une nouvelle valeur qui doit être typée.

\vspace{0.2cm}
\item Dans l'exemple, le \textbf{\textcolor{NavyBlue}{type}} est \texttt{\textcolor{NavyBlue}{int[]}}, les indices de \texttt{\textcolor{NavyBlue}{0}} à \texttt{\textcolor{NavyBlue}{6}} correspondent  aux jours de la semaine.

\vspace{0.2cm}
\item On peut alors déclarer une variable (\texttt{uneSemaine}) de ce nouveau type, enregistrer des valeurs dans les cases de ce tableau, et sélectionner une case connaissant son indice (\texttt{uneSemaine[i]} ).

\end{itemize}
\end{frame}



\begin{frame}[fragile]\frametitle{Les tableaux en \texttt{C}}


  \begin{itemize}
  \item C'est une variable qui contient \emph{\textcolor{NavyBlue}{plusieurs valeurs}} d'un même  type.
  \vspace{0.2cm}
  \item Si il a \texttt{\textcolor{NavyBlue}{n}} valeurs, elles ont  \emph{numérotées de $0$ à $n-1$} \danger
   \vspace{0.2cm}
  \item la i$^e$ case d'un tableau \lstinlineCv!t!, (notée
    \lstinlineCv!t[i]!) = équivalente à une variable.
     \vspace{0.1cm}
    \begin{itemize}
    \item \textcolor{NavyBlue}{Affectation :} \lstinlineCv!t[i] = 12 + f(x);!
        \vspace{0.1cm}
    \item \textcolor{NavyBlue}{Utilisation :} \lstinlineCv!x = (t[i] - 1) * u[j];!
    \end{itemize}
     \vspace{0.2cm}
  \item \textcolor{NavyBlue}{Déclaration :}
    \begin{lstlistingC}[]
int t [9]; // 9 cases
int t = {9,8,7,6,5,4,3,2,1}; //9 cases
\end{lstlistingC}
\end{itemize}
\end{frame}

\begin{frame}[fragile]\frametitle{Parcours d'un tableau}
    \begin{lstlistingC}[]
int t [9]; // 9 cases
int t = {9,8,7,6,5,4,3,2,1}; //9 cases
int idx = 0;
while(idx</*@\tikz \coordinate (lgthwhile);@*/9) {
  t[idx] = 0;
  idx = idx + 1;
}
/*@\tikz \coordinate (afterwhile);@*/
for(idx=0; idx</*@\tikz \coordinate (lgthfor);@*/9 ; idx = idx + 1){
  t[idx] = idx;
}
/*@\tikz \coordinate (afterfor);@*/

\begin{frame}[fragile]\frametitle{Rappel : le typage}
  \begin{center}
    \begin{minipage}{.7\linewidth}
\begin{lstlistingC}
int x;
/*@\tikz\coordinate(aff);@*/x = 8;
... /*@\tikz\coordinate(op);@*/(x + 9) ...;
... /*@\tikz\coordinate(func);@*/f(x) ...;/*@\hfill@*/ /* int f (int a) */
\end{lstlistingC}
    \end{minipage}
  \end{center}
  \begin{itemize}
  \item Chaque variable est déclarée avec son type \texttt{\textcolor{NavyBlue}{T}} (\lstinlineCv!int!, \lstinlineCv!char!, ...)
  \vspace{0.1cm}
  \item<2-> \textcolor{NavyBlue}{Affectation: }seulement par une valeur du type  \texttt{\textcolor{NavyBlue}{T}}
   \vspace{0.1cm}
  \item<3-> \textcolor{NavyBlue}{Opérateur : }seulement si  \texttt{\textcolor{NavyBlue}{T}} est le bon type
   \vspace{0.1cm}
  \item<4-> \textcolor{NavyBlue}{Fonction :} seulement si  \texttt{\textcolor{NavyBlue}{T}} est le bon type
  
    \end{itemize}
     \vspace{0.2cm}
   $\textcolor{NavyBlue}{\Rightarrow}$ Sécurité, pas d'erreur de type à l'exécution \\
   
    \vspace{0.2cm}
   \textcolor{NavyBlue}{MAIS:} conversion entre type (pratique mais source d'erreurs)

  \begin{tikzpicture}[remember picture, overlay]
    \small\tt%
    \node<all:2|handout:1>[ovalise=width("x x 8;")] (ovalaff) at (aff) {};
    \node<all:3|handout:1>[ovalise=width("(x + 9)")] (ovalop) at (op) {};
    \node<all:4|handout:1>[ovalise=width("f(x)")] (ovalfunc) at (func) {};
  \end{tikzpicture}
\end{frame}
