\begin{lstlisting}
void main(){
	int a,i,compt;
	i=0;
	compt=0;
	ecrireString("Entrez un nombre");
	a=lireInt();
	while(i<=a) {
		if(i%2==0){
			ecrireString(" "+i);
			ecrireSautDeLigne();
			compt=compt+1;
		}
		i=i+1;
	}
	ecrireString("En tout "+ compt + " entiers.");
}
\end{lstlisting}
